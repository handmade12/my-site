Pro Micro BadUSB HID

BadUSB на Pro Micro

Arduino Pro Micro притворяется USB-клавиатурой. Открывает PowerShell, выводит ASCII-анимацию Never Gonna Give You Up, показывает сообщения в txt, запускает ссылки.

C++ (Arduino Pro Micro) 
// BadUSB — клавиатурная эмуляция на Pro Micro
#include <Keyboard.h>

void setup() {
  Keyboard.begin();
  delay(1000);

  // Win + R → powershell
  Keyboard.press(KEY_LEFT_GUI);
  Keyboard.press('r');
  delay(300);
  Keyboard.releaseAll();
  delay(500);
  Keyboard.print("powershell");
  Keyboard.press(KEY_RETURN);
  Keyboard.release(KEY_RETURN);
  delay(1500);

  // ASCII-анимация Rickroll
  Keyboard.print("echo 'Never Gonna Give You Up'");
  Keyboard.press(KEY_RETURN);
  Keyboard.release(KEY_RETURN);
  delay(200);
  Keyboard.print("echo '  ╔══╗╔══╗╔╗─╔╗╔══╗╔══╗'");
  Keyboard.press(KEY_RETURN);
  Keyboard.release(KEY_RETURN);
  delay(200);
  Keyboard.print("echo '  ╚╗╔╝║╔═╝║║─║║║╔╗║║╔═╝'");
  Keyboard.press(KEY_RETURN);
  Keyboard.release(KEY_RETURN);

  // Создаём txt-файл с сообщением
  Keyboard.print("echo 'MRX был тут ;)' > Desktop\\hello.txt");
  Keyboard.press(KEY_RETURN);
  Keyboard.release(KEY_RETURN);

  // Открываем видео
  Keyboard.print("Start https://youtu.be/dQw4w9WgXcQ");
  Keyboard.press(KEY_RETURN);
  Keyboard.release(KEY_RETURN);
}

void loop() {}
Arduino Звук

Кузнечик — мелодия на пищалке

Первый проект после покупки набора. Пьезопищалка (buzzer) играет мелодию «В траве сидел кузнечик» по нажатию кнопки. База, но с неё всё началось.

C++ (Arduino) 
// Кузнечик — мелодия на пьезопищалке
const int BUZZER = 9;
const int BUTTON = 2;

// Частоты нот мелодии
int melody[] = {
  262, 262, 392, 392, 440, 440, 392,
  349, 349, 330, 330, 294, 294, 262
};
int noteDurations[] = {
  200, 200, 200, 200, 200, 200, 400,
  200, 200, 200, 200, 200, 200, 400
};

void setup() {
  pinMode(BUZZER, OUTPUT);
  pinMode(BUTTON, INPUT_PULLUP);
}

void loop() {
  if (digitalRead(BUTTON) == LOW) {
    playMelody();
  }
}

void playMelody() {
  for (int i = 0; i < 14; i++) {
    tone(BUZZER, melody[i], noteDurations[i]);
    delay(noteDurations[i] + 50);
    noTone(BUZZER);
  }
}
