Pro Micro Тест Leonardo

Pro Micro: проверка связи

Базовый тест платы Pro Micro. Мигает встроенным светодиодом. В Arduino IDE нужно выбрать плату Arduino Leonardo — Pro Micro нет в списке, но они совместимы.

C++ (Arduino Pro Micro) 
void setup() {
  pinMode(LED_BUILTIN, OUTPUT);
}

void loop() {
  digitalWrite(LED_BUILTIN, HIGH);
  delay(500);
  digitalWrite(LED_BUILTIN, LOW);
  delay(500);
}
Arduino Звук

Кузнечик — мелодия на пищалке

Первый проект после покупки набора. Пьезопищалка (buzzer) играет мелодию «В траве сидел кузнечик» по нажатию кнопки. База, но с неё всё началось.

C++ (Arduino) 
const int BUZZER = 9;
const int BUTTON = 2;

int melody[] = {
  262, 262, 392, 392, 440, 440, 392,
  349, 349, 330, 330, 294, 294, 262
};
int noteDurations[] = {
  200, 200, 200, 200, 200, 200, 400,
  200, 200, 200, 200, 200, 200, 400
};

void setup() {
  pinMode(BUZZER, OUTPUT);
  pinMode(BUTTON, INPUT_PULLUP);
}

void loop() {
  if (digitalRead(BUTTON) == LOW) {
    playMelody();
  }
}

void playMelody() {
  for (int i = 0; i < 14; i++) {
    tone(BUZZER, melody[i], noteDurations[i]);
    delay(noteDurations[i] + 50);
    noTone(BUZZER);
  }
}
Pro Micro BadUSB CMD

BadUSB: Rick Roll в CMD

Pro Micro через Win+R запускает cmd, разворачивает окно, красит фон в хакерский зелёный и выводит ASCII-анимацию.

C++ (Arduino Pro Micro) 
#include <Keyboard.h>

void setup() {
  delay(5000);
  Keyboard.begin();
  delay(1000);

  Keyboard.press(KEY_LEFT_GUI);
  Keyboard.press('r');
  delay(100);
  Keyboard.releaseAll();
  delay(500);

  Keyboard.print("cmd");
  delay(200);
  Keyboard.press(KEY_RETURN);
  delay(100);
  Keyboard.releaseAll();
  delay(1000);

  Keyboard.print("mode con: cols=200 lines=60");
  Keyboard.press(KEY_RETURN);
  delay(100);
  Keyboard.releaseAll();
  delay(200);
  Keyboard.print("color 0a");
  Keyboard.press(KEY_RETURN);
  delay(100);
  Keyboard.releaseAll();

  Keyboard.print("tree c:\\");
  Keyboard.press(KEY_RETURN);

  Keyboard.end();
}

void loop() {}
Pro Micro BadUSB PowerShell

BadUSB: сообщение в .txt

Pro Micro через PowerShell создаёт .txt файл с кастомным сообщением и открывает его в Блокноте — быстро и без лишних окон.

C++ (Arduino Pro Micro) 
#include "Keyboard.h"

void setup() {
  Keyboard.begin();
  delay(2000);

  Keyboard.press(KEY_LEFT_GUI);
  Keyboard.press('r');
  delay(50);
  Keyboard.releaseAll();
  delay(200);

  Keyboard.print("powershell -W Hidden -C \"'hello', 'I''m here'|Set-Content $env:TEMP\\n.txt;Start-Process notepad $env:TEMP\\n.txt\"");
  Keyboard.press(KEY_RETURN);
  delay(50);
  Keyboard.release(KEY_RETURN);

  Keyboard.end();
}

void loop() {}
Pro Micro BadUSB YouTube

BadUSB: открывает YouTube

Pro Micro через Win+R запускает cmd, который открывает браузер с видео на YouTube.

C++ (Arduino Pro Micro) 
#include "Keyboard.h"

void setup() {
  Keyboard.begin();
  delay(3000);

  Keyboard.press(KEY_LEFT_GUI);
  Keyboard.press('r');
  delay(100);
  Keyboard.releaseAll();
  delay(500);

  Keyboard.print("cmd /c start https://www.youtube.com/watch?v=Qs4OcQNOM34");
  Keyboard.press(KEY_RETURN);
  Keyboard.release(KEY_RETURN);

  Keyboard.end();
}

void loop() {}
